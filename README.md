> A Vue.js 2.0 UI Toolkit for Web.


## Install
```shell
npm install kyligence-ui -S
```

## Quick Start
``` javascript
import Vue from 'vue'
import KyligenceUI from 'kyligence-ui'

Vue.use(KyligenceUI)

// or
import {
  Select,
  Button
  // ...
} from 'kyligence-ui'

Vue.component(Select.name, Select)
Vue.component(Button.name, Button)
```
